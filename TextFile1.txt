using System;
using Xunit;
using FluentAssertions;
using GertToUTW;

namespace GertToUTW.Tests;

public class TestItemTests
{
    #region Idx Property Tests

    [Theory]
    [InlineData(0)]
    [InlineData(1)]
    [InlineData(16384)]
    [InlineData(32767)]
    [InlineData(null)]
    public void Idx_WithValidValue_ShouldSetProperty(int? validValue)
    {
        // Arrange
        var item = new TestItem();

        // Act
        item.Idx = validValue;

        // Assert
        item.Idx.Should().Be(validValue);
    }

    [Theory]
    [InlineData(-1)]
    [InlineData(32768)]
    [InlineData(int.MinValue)]
    [InlineData(int.MaxValue)]
    public void Idx_WithInvalidValue_ShouldThrowArgumentOutOfRangeException(int invalidValue)
    {
        // Arrange
        var item = new TestItem();

        // Act
        Action act = () => item.Idx = invalidValue;

        // Assert
        act.Should().Throw<ArgumentOutOfRangeException>()
           .WithParameterName("value")
           .WithMessage("*Idx must be a valid positive short integer (0 to 32767).*");
    }

    #endregion

    #region Result Property Tests

    [Theory]
    [InlineData("PASSED", "PASSED")]
    [InlineData("failed", "FAILED")] // Lowercase check
    [InlineData("Skipped", "SKIPPED")] // Mixed-case check
    [InlineData("INCOMPLETE", "INCOMPLETE")]
    [InlineData("ERROR", "ERROR")]
    public void Result_WithValidValue_ShouldSetToUppercase(string inputValue, string expectedValue)
    {
        // Arrange
        var item = new TestItem();

        // Act
        item.Result = inputValue;

        // Assert
        item.Result.Should().Be(expectedValue);
    }

    [Fact]
    public void Result_WithNullValue_ShouldSetToEmptyString()
    {
        // Arrange
        var item = new TestItem();

        // Act
        item.Result = null!; // Using ! to bypass nullable warning for the test case

        // Assert
        item.Result.Should().Be(string.Empty);
    }

    [Theory]
    [InlineData("PASS")]
    [InlineData("FAIL")]
    [InlineData("WRONG_STATUS")]
    [InlineData(" ")]
    public void Result_WithInvalidValue_ShouldThrowArgumentException(string invalidValue)
    {
        // Arrange
        var item = new TestItem();

        // Act
        Action act = () => item.Result = invalidValue;

        // Assert
        act.Should().Throw<ArgumentException>()
           .WithParameterName("value")
           .WithMessage($"*Invalid Result '{invalidValue}'.*");
    }

    #endregion

    #region Basic Property Tests

    [Fact]
    public void AutoProperties_ShouldGetAndSetCorrectly()
    {
        // Arrange
        var item = new TestItem();

        // Act
        item.TestItem_Key = 123;
        item.Name = "Test Name";
        item.Description = "Test Description";
        item.Stdout = "Standard Output Log";
        item.Stderr = "Standard Error Log";

        // Assert
        item.TestItem_Key.Should().Be(123);
        item.Name.Should().Be("Test Name");
        item.Description.Should().Be("Test Description");
        item.Stdout.Should().Be("Standard Output Log");
        item.Stderr.Should().Be("Standard Error Log");
    }

    #endregion
}
